Functions & Returns

Beginner
Master the signature → logic → return. Useful for greet, add, greetName, nextInteger, etc. 
function greet(name){ 
                return "Hello, " + name + "!";
                }

                function add(a,b){ 
                return a + b; 
                }

                function nextInteger(n){ 
                return n + 1; 
                }
• Square: n*n
• Double/Triple: n*2, n*3
• Half: n/2
• C→F: c*9/5+32

String Algorithms

Intermediate
Reversal, vowel counting, palindrome, anagram, uniqueness, most-frequent. 
function reverseString(s){ return s.split("").reverse().join(""); }
                function countVowels(s){ const v="aeiouAEIOU"; return [...s].filter(ch=>v.includes(ch)).length; }
                function isPalindrome(s){ const t=s.toLowerCase(); return t===reverseString(t); }
                function isAnagram(a,b){
                const norm = s => s.toLowerCase().replace(/\s+/g,"").split("").sort().join("");
                return norm(a)===norm(b);
                }
                function allUnique(s){ const seen=new Set(); for(const ch of s){ if(seen.has(ch)) return false; seen.add(ch); } return true; }
                function mostFrequentChar(s){
                const f={}; let best="",cnt=0;
                for(const ch of s){ f[ch]=(f[ch]||0)+1; if(f[ch]>cnt){ cnt=f[ch]; best=ch; } }
                return best;
                }

Arrays: Search & Math Patterns

Intermediate
Index finding, two-sum with a set, second largest, reduce sum. 
function findIndex(arr,val){ for(let i=0;i<arr.length;i++) if(arr[i]===val) return i; return -1; }
                function sumArray(arr){ return arr.reduce((s,x)=>s+x,0); }
                function hasTwoSum(arr,target){
                const seen=new Set();
                for(const x of arr){ if(seen.has(target-x)) return true; seen.add(x); }
                return false;
                }
                function secondLargest(arr){
                let max=-Infinity, second=-Infinity;
                for(const x of arr){ if(x>max){ second=max; max=x; } else if(x>second){ second=x; } }
                return second;
                }
Number theory helpers 
function gcd(a,b){ while(b!==0){ [a,b]=[b,a%b]; } return a; }
                function isPrime(n){
                if(n<2) return false; if(n%2===0) return n===2;
                for(let i=3;i*i<=n;i+=2) if(n%i===0) return false;
                return true;
                }

Stacks & Parentheses

Intermediate
Balance counting for parentheses (no extra memory needed here). 
function isBalanced(s){
                let c=0;
                for(const ch of s){
                    if(ch==="(") c++;
                    else if(ch===")"){ c--; if(c<0) return false; }
                }
                return c===0;
                }

DP & Classic Algorithms

Advanced
Use these canonical implementations for the advanced set.
1) Fibonacci / Climb Stairs 
function fibonacci(n){ if(n<=1) return n; let a=0,b=1; for(let i=2;i<=n;i++) [a,b]=[b,a+b]; return b; }
                function climbStairs(n){ if(n<=2) return n; let a=1,b=2; for(let i=3;i<=n;i++) [a,b]=[b,a+b]; return b; }
2) Kadane (Max Subarray) 
function maxSubarraySum(arr){
                let best=-Infinity, cur=0;
                for(const x of arr){ cur=Math.max(x,cur+x); best=Math.max(best,cur); }
                return best;
                }
3) LIS (Patience w/ Binary Search) 
function lisLength(arr){
                const d=[];
                for(const x of arr){
                    let l=0,r=d.length;
                    while(l<r){
                    const m=(l+r)>>1;
                    if(d[m]<x) l=m+1; else r=m;
                    }
                    d[l]=x;
                }
                return d.length;
                }
4) Count Inversions (Merge) 
function countInversions(a){
                function go(arr){
                    if(arr.length<2) return {a:[...arr], c:0};
                    const m=arr.length>>1, L=go(arr.slice(0,m)), R=go(arr.slice(m));
                    let i=0,j=0,c=L.c+R.c,out=[];
                    while(i<L.a.length && j<R.a.length){
                    if(L.a[i] <= R.a[j]) out.push(L.a[i++]);
                    else { out.push(R.a[j++]); c += L.a.length - i; }
                    }
                    while(i<L.a.length) out.push(L.a[i++]);
                    while(j<R.a.length) out.push(R.a[j++]);
                    return {a:out, c};
                }
                return go(a).c;
                }
5) Quickselect (k-th smallest) 
function kthSmallest(arr,k){
                k--; const a=[...arr];
                function part(lo,hi){
                    const p=a[hi]; let i=lo;
                    for(let j=lo;j<hi;j++) if(a[j]<=p) [a[i],a[j]]=[a[j],a[i]], i++;
                    [a[i],a[hi]]=[a[hi],a[i]]; return i;
                }
                let l=0,r=a.length-1;
                while(l<=r){ const p=part(l,r); if(p===k) return a[p]; if(p<k) l=p+1; else r=p-1; }
                }
6) Primes: sieve, nth prime, count ≤ n 
function countPrimesUpTo(n){
                if(n<2) return 0; const s=new Uint8Array(n+1); let c=0;
                for(let i=2;i<=n;i++){ if(!s[i]){ c++; if(i*i<=n) for(let j=i*i;j<=n;j+=i) s[j]=1; } }
                return c;
                }
function nthPrime(n){
                if(n===1) return 2;
                const est = n<6 ? 15 : Math.ceil(n*(Math.log(n)+Math.log(Math.log(n))));
                const limit = Math.max(est,15);
                const sieve = new Uint8Array(limit+1);
                let cnt=0;
                for(let i=2;i<=limit;i++){
                    if(!sieve[i]){
                    if(++cnt===n) return i;
                    for(let j=i*i;j<=limit;j+=i) sieve[j]=1;
                    }
                }
                // expand if estimate low
                let size=limit*2;
                while(true){
                    const s=new Uint8Array(size+1);
                    for(let p=2;p*p<=size;p++) if(!s[p]) for(let j=p*p;j<=size;j+=p) s[j]=1;
                    cnt=0; for(let i=2;i<=size;i++) if(!s[i]) if(++cnt===n) return i;
                    size*=2;
                }
                }
function modPow(base, exp, m){
                base%=m; let res=1%m;
                while(exp>0){ if(exp&1) res=(res*base)%m; base=(base*base)%m; exp>>=1; }
                return res;
                }
7) Edit Distance (Levenshtein) 
function editDistance(a,b){
                const n=a.length,m=b.length, dp=Array(n+1).fill(0).map(()=>Array(m+1).fill(0));
                for(let i=0;i<=n;i++) dp[i][0]=i; for(let j=0;j<=m;j++) dp[0][j]=j;
                for(let i=1;i<=n;i++) for(let j=1;j<=m;j++)
                    dp[i][j] = a[i-1]===b[j-1] ? dp[i-1][j-1] : 1+Math.min(dp[i-1][j],dp[i][j-1],dp[i-1][j-1]);
                return dp[n][m];
                }
8) 0/1 Knapsack (1D DP) 
function knapsack01(W,V,C){
                const n=W.length, dp=new Array(C+1).fill(0);
                for(let i=0;i<n;i++) for(let c=C;c>=W[i];c--) dp[c]=Math.max(dp[c], dp[c-W[i]]+V[i]);
                return dp[C];
                }
